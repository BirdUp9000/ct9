194[[nodiscard]] std::queue<std::string> Trie::Node::autocompleteNode(const std::string& prefix, size_t count) const {
195 std::stack<std::pair<const Node*, std::string>> stack;
196 std::queue<std::string> results;
197 stack.push({this, prefix});
198
199 while (!stack.empty() && results.size() < count) {
200 auto [current_node, word] = stack.top();
201 stack.pop();
202
203 if (current_node->end_of_word) {
204 results.push(word);
205 }
206
207 for (auto it = current_node->children.rbegin(); it != current_node->children.rend(); ++it) {
208 stack.push({it->second, word + it->first});
209 }
210 }
211
212 return results;
213}

237inline void Trie::Node::insert(const std::string& text, const size_t index, Node* root) {
238 if (index == text.size()) {
239 end_of_word = true;
240 return;
241 }
242
243 // Mark as end of word if a non-alphabetic character is encountered
244 if (isalpha(text[index]) == 0) {
245 end_of_word = true;
246 root->insert(text, index + 1, root);
247 } else {
248 // Create a new node if the current character is not found
249 if (!children.contains(text[index])) {
250 children[text[index]] = new Node();

252 children[text[index]]->insert(text, index + 1, root);
253 }
