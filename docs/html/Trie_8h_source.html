139void Trie::del(const std::string& word) {
140 bool is_word = this->autocomplete(word, 1).size() == 1;
141 if (!is_word) return;
142 Node* tmp = root;
143 std::vector<Node*> path;
144 for (char c : word) {
145 path.push_back(tmp);
146 if (tmp->children.find(c) == tmp->children.end()) return;
147 tmp = tmp->children[c];
148 }
149
150 tmp->end_of_word = false;
151
152 for (int i = path.size() - 1; i >= 0; --i) {
153 Node* parent = path[i];
154 char c = word[i];
155
156 if (parent->children[c]->children.empty() && !parent->children[c]->end_of_word) {
157 delete parent->children[c];
158 parent->children.erase(c);
159 } else {
160 break;
161 }
162 }

226[[nodiscard]] std::queue<std::string> Trie::Node::autocompleteNode(const std::string& prefix, size_t count) const {
227 std::stack<std::pair<const Node*, std::string>> stack;
228 std::queue<std::string> results;
229 stack.push({this, prefix});
230
231 while (!stack.empty() && results.size() < count) {
232 auto [current_node, word] = stack.top();
233 stack.pop();
234
235 if (current_node->end_of_word) {
236 results.push(word);
237 }
238
239 for (auto it = current_node->children.rbegin(); it != current_node->children.rend(); ++it) {
240 stack.push({it->second, word + it->first});
241 }
242 }
243
244 return results;
245}

269inline void Trie::Node::insert(const std::string& text, const size_t index, Node* root) {
