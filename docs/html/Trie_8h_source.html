83[[nodiscard]] inline std::queue<std::string> Trie::Node::autocompleteNode(const std::string& prefix) const {
84 std::queue<std::string> results;
85 std::function<void(const Node*, std::string)> dfs = [&](const Node* node, const std::string& current) {
86 if (node->end_of_word) { results.push(current); };
87 for (const auto& [key, child] : node->children) {
88 dfs(child, current + key);
89 }

91 dfs(this, prefix);
