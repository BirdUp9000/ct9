285inline void Trie::copyNodes(Node* dstRoot, const Node* srcRoot) {
286 dstRoot->clearNode();
287 std::queue<const Node*> srcQueue;
288 std::queue<Node*> dstQueue;
289
290 srcQueue.push(srcRoot);
291 dstQueue.push(dstRoot);
292
293 while (!srcQueue.empty()) {
294 auto currentSrcNode = srcQueue.front();
295 srcQueue.pop();
296
297 auto currentDstNode = dstQueue.front();
298 dstQueue.pop();
299
300 for (auto& child : currentSrcNode->children) {
301 srcQueue.push(child.second);
302
303 char childChar = child.first;
304 Node* newDstNode = new Node;
305 newDstNode->end_of_word = child.second->end_of_word;
306
307 currentDstNode->children[childChar] = newDstNode;
308 dstQueue.push(newDstNode);
309 }
310 }

322inline void Trie::del(const std::string& word) const {
323 bool is_word = this->autocomplete(word, 1).size() == 1;
324 if (!is_word) {
325 return;
326 }
327 Node* tmp = root;
328 std::vector<Node*> path;
329 for (char c : word) {
330 path.push_back(tmp);
331 if (tmp->children.find(c) == tmp->children.end()) {
332 return;
333 }
334 tmp = tmp->children[c];
335 }
336
337 tmp->end_of_word = false;
338
339 for (int i = path.size() - 1; i >= 0; --i) {
340 Node* parent = path[i];
341 char c = word[i];
342
343 if (parent->children[c]->children.empty() && !parent->children[c]->end_of_word) {
344 delete parent->children[c];
345 parent->children.erase(c);
346 } else {
347 break;
348 }
349 }
350}

433[[nodiscard]] inline std::queue<std::string> Trie::Node::autocompleteNode(const std::string& prefix,
434 const size_t count) const {
435 std::queue<std::string> results;
436 std::string current = prefix;
437
438 std::function<bool(const Node*)> dfs = [&](const Node* node) -> bool {
439 if (results.size() >= count) {
440 return true;
441 }
442
443 if (node->end_of_word) {
444 results.push(current);
445 if (results.size() >= count) {
446 return true;
447 }
448 }
449
450 for (const auto& [key, child] : node->children) {
451 current.push_back(key);
452 if (dfs(child)) {
453 return true;
454 }
455 current.pop_back();
456 }
457 return false;
458 };
459
460 static_cast<void>(dfs(this));
461 return results;
462}

484inline void Trie::Node::insert(const std::string& text, const size_t index, Node* root) {
485 if (index == text.size()) {
486 end_of_word = true;
487 return;
488 }
489
490 // Mark as end of word if a non-alphabetic character is encountered
491 if (isalpha(text[index]) == 0) {
492 end_of_word = true;
493 root->insert(text, index + 1, root);
494 } else {
495 // Create a new node if the current character is not found
496 if (!children.contains(text[index])) {
497 children[text[index]] = new Node();
498 }
499 children[text[index]]->insert(text, index + 1, root);
500 }
