137inline void Trie::copyNodes(Node* dstRoot, const Node* srcRoot) {
138 dstRoot->clearNode();
139 std::queue<const Node*> srcQueue;
140 std::queue<Node*> dstQueue;
141
142 srcQueue.push(srcRoot);
143 dstQueue.push(dstRoot);
144
145 while (!srcQueue.empty()) {
146 auto currentSrcNode = srcQueue.front();
147 srcQueue.pop();
148
149 auto currentDstNode = dstQueue.front();
150 dstQueue.pop();
151
152 for (auto& child : currentSrcNode->children) {
153 srcQueue.push(child.second);
154
155 char childChar = child.first;
156 Node* newDstNode = new Node;
157 newDstNode->end_of_word = child.second->end_of_word;
158
159 currentDstNode->children[childChar] = newDstNode;
160 dstQueue.push(newDstNode);
161 }
162 }

174inline void Trie::del(const std::string& word) const {
175 bool is_word = this->autocomplete(word, 1).size() == 1;
176 if (!is_word) {return;}
177 Node* tmp = root;
178 std::vector<Node*> path;
179 for (char c : word) {
180 path.push_back(tmp);
181 if (tmp->children.find(c) == tmp->children.end()) {return;}
182 tmp = tmp->children[c];
183 }
184
185 tmp->end_of_word = false;
186
187 for (int i = path.size() - 1; i >= 0; --i) {
188 Node* parent = path[i];
189 char c = word[i];
190
191 if (parent->children[c]->children.empty() && !parent->children[c]->end_of_word) {
192 delete parent->children[c];
193 parent->children.erase(c);
194 } else {
195 break;
196 }
197 }

283[[nodiscard]] inline std::queue<std::string> Trie::Node::autocompleteNode(const std::string& prefix, const size_t count) const {
284 std::queue<std::string> results;
285 std::string current = prefix;
286
287 std::function<bool(const Node*)> dfs = [&](const Node* node) -> bool {
288 if (results.size() >= count) {return true;}
289
290 if (node->end_of_word) {
291 results.push(current);
292 if (results.size() >= count) {return true;}
293 }
294
295 for (const auto& [key, child] : node->children) {
296 current.push_back(key);
297 if (dfs(child)) {return true;}
298 current.pop_back();
299 }
300 return false;
301 };
302
303 static_cast<void>(dfs(this));
304 return results;
305}

329inline void Trie::Node::insert(const std::string& text, const size_t index, Node* root) {
330 if (index == text.size()) {
331 end_of_word = true;
332 return;
333 }
334
335 // Mark as end of word if a non-alphabetic character is encountered
336 if (isalpha(text[index]) == 0) {
337 end_of_word = true;
338 root->insert(text, index + 1, root);
339 } else {
340 // Create a new node if the current character is not found
341 if (!children.contains(text[index])) {
342 children[text[index]] = new Node();
343 }
344 children[text[index]]->insert(text, index + 1, root);
345 }
