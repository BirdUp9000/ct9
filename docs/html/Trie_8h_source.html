188inline void Trie::copyNodes(Node* dstRoot, const Node* srcRoot) {
189 dstRoot->clearNode();
190 std::queue<const Node*> srcQueue;
191 std::queue<Node*> dstQueue;
192
193 srcQueue.push(srcRoot);
194 dstQueue.push(dstRoot);
195
196 while (!srcQueue.empty()) {
197 auto currentSrcNode = srcQueue.front();
198 srcQueue.pop();
199
200 auto currentDstNode = dstQueue.front();
201 dstQueue.pop();
202
203 for (auto& child : currentSrcNode->children) {
204 srcQueue.push(child.second);
205
206 char childChar = child.first;
207 Node* newDstNode = new Node;
208 newDstNode->end_of_word = child.second->end_of_word;
209
210 currentDstNode->children[childChar] = newDstNode;
211 dstQueue.push(newDstNode);
212 }
213 }

225inline void Trie::del(const std::string& word) const {
226 bool is_word = this->autocomplete(word, 1).size() == 1;
227 if (!is_word) {
228 return;
229 }
230 Node* tmp = root;
231 std::vector<Node*> path;
232 for (char c : word) {
233 path.push_back(tmp);
234 if (tmp->children.find(c) == tmp->children.end()) {
235 return;
236 }
237 tmp = tmp->children[c];
238 }
239
240 tmp->end_of_word = false;
241
242 for (int i = path.size() - 1; i >= 0; --i) {
243 Node* parent = path[i];
244 char c = word[i];
245
246 if (parent->children[c]->children.empty() && !parent->children[c]->end_of_word) {
247 delete parent->children[c];
248 parent->children.erase(c);
249 } else {
250 break;
251 }
252 }
253}

336[[nodiscard]] inline std::queue<std::string> Trie::Node::autocompleteNode(const std::string& prefix,
337 const size_t count) const {
338 std::queue<std::string> results;
339 std::string current = prefix;
340
341 std::function<bool(const Node*)> dfs = [&](const Node* node) -> bool {
342 if (results.size() >= count) {
343 return true;
344 }
345
346 if (node->end_of_word) {
347 results.push(current);
348 if (results.size() >= count) {
349 return true;
350 }
351 }
352
353 for (const auto& [key, child] : node->children) {
354 current.push_back(key);
355 if (dfs(child)) {
356 return true;
357 }
358 current.pop_back();
359 }
360 return false;
361 };
362
363 static_cast<void>(dfs(this));
364 return results;
365}

387inline void Trie::Node::insert(const std::string& text, const size_t index, Node* root) {
388 if (index == text.size()) {
389 end_of_word = true;

391 }

398 // Create a new node if the current character is not found
