251inline void Trie::copyNodes(Node* dstRoot, const Node* srcRoot) {
252 dstRoot->clearNode();
253 std::queue<const Node*> srcQueue;
254 std::queue<Node*> dstQueue;
255
256 srcQueue.push(srcRoot);
257 dstQueue.push(dstRoot);
258
259 while (!srcQueue.empty()) {
260 auto currentSrcNode = srcQueue.front();
261 srcQueue.pop();
262
263 auto currentDstNode = dstQueue.front();
264 dstQueue.pop();
265
266 for (auto& child : currentSrcNode->children) {
267 srcQueue.push(child.second);

269 char childChar = child.first;
270 Node* newDstNode = new Node;
271 newDstNode->end_of_word = child.second->end_of_word;
272
273 currentDstNode->children[childChar] = newDstNode;
274 dstQueue.push(newDstNode);

288inline void Trie::del(const std::string& word) const {
289 bool is_word = this->autocomplete(word, 1).size() == 1;
290 if (!is_word) {
291 return;
292 }
293 Node* tmp = root;
294 std::vector<Node*> path;
295 for (char c : word) {
296 path.push_back(tmp);
297 if (tmp->children.find(c) == tmp->children.end()) {
298 return;
299 }
300 tmp = tmp->children[c];
301 }
302
303 tmp->end_of_word = false;
304
305 for (int i = path.size() - 1; i >= 0; --i) {
306 Node* parent = path[i];
307 char c = word[i];
308
309 if (parent->children[c]->children.empty() && !parent->children[c]->end_of_word) {
310 delete parent->children[c];
311 parent->children.erase(c);
312 } else {
313 break;
314 }
315 }

399[[nodiscard]] inline std::queue<std::string> Trie::Node::autocompleteNode(const std::string& prefix,
400 const size_t count) const {
401 std::queue<std::string> results;
402 std::string current = prefix;
403
404 std::function<bool(const Node*)> dfs = [&](const Node* node) -> bool {
405 if (results.size() >= count) {
406 return true;
407 }
408
409 if (node->end_of_word) {
410 results.push(current);
411 if (results.size() >= count) {
412 return true;
413 }
414 }
415
416 for (const auto& [key, child] : node->children) {
417 current.push_back(key);
418 if (dfs(child)) {
419 return true;
420 }
421 current.pop_back();
422 }
423 return false;
424 };
425
426 static_cast<void>(dfs(this));
427 return results;
428}

450inline void Trie::Node::insert(const std::string& text, const size_t index, Node* root) {
451 if (index == text.size()) {
452 end_of_word = true;
453 return;
454 }
455
456 // Mark as end of word if a non-alphabetic character is encountered
457 if (isalpha(text[index]) == 0) {
458 end_of_word = true;
459 root->insert(text, index + 1, root);
460 } else {
461 // Create a new node if the current character is not found
462 if (!children.contains(text[index])) {
463 children[text[index]] = new Node();
464 }
465 children[text[index]]->insert(text, index + 1, root);
466 }
